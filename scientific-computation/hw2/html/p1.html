%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% [ 0.5  0.25  0.125  ]    [ a2 ]    [ y ]
% [  1     1     1    ]    [ a3 ]  = [ 3 ]
% [  2     4     8    ]    [ 6  ]    [ 2 ]
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

M = [0.5 0.25 0.125; 1 1 1; 2 4 8];
t = [a2 a3 6]';
rs = [y 3 2]';

% the solve(...) function will solve symbolic equations
% see help solve
sol = solve(M * t == r);
disp('the results are: \n')
fprintf('%3.3f, %3.3f, %3.3f, %3.3f, %3.3f', 0.00, sol.a2, sol.a3, 6.00, sol.y)
